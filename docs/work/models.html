Tests de modèles

Tous les tests réalisés seront effectués avec une exposition topographique au vent calculé avec un angle de 6°

On peut faire une première visualisation simple des données afin de savoir comment les variables se comportent entre elles.

# donnees ####
dta <- read.csv("~/Documents/scriptR tests/datasummary.csv", sep = ",", header = TRUE)
dta <- dta[ ,c("ang6","pente","longitude","latitude","altitude","courbure","msw_max","delta_ndvi","wem","delta_ndii")]
# découpage des données en un jeu d'entrainement (70%) et un jeu de test (30%)
set.seed(1234)

dta_test <- dta[-train_id, ]

msw_range <- seq(min(dta$msw_max), max(dta$msw_max), length.out = 300)

# ---- Fonction utilitaire R² ----
r2 <- function(obs, pred) {
  1 - sum((obs - pred)^2) / sum((obs - mean(obs))^2)
}

# ============================================================
# 1. MODÈLE LINÉAIRE
# ============================================================
m1 <- lm(delta_ndvi ~ msw_max, data = dta_train)
r2_m1_train <- r2(dta_train$delta_ndvi, predict(m1, dta_train))
r2_m1_test  <- r2(dta_test$delta_ndvi,  predict(m1, dta_test))
